\section{Auslegung Antenne}
\begin{frame}
    \frametitle{Auswahl einer Bauweise}
    \begin{itemize}
        \item Häufig in käuflichen Empfänger-Modulen genutzt~\cite{piester_zeit-_2004}
        \item kostengünstig in der Beschaffung
        \item einfache Produktion
    \end{itemize}
    \vspace{2em}
    {\large \bfseries \(\Rightarrow\) Ferrit-Stab Antenne}
\end{frame}
\begin{frame}
    \frametitle{Berechnung einer Ferritantenne}
    \begin{block}{Frage:}
        Wie kann die Antenne auf die Trägerfrequenz eingestellt werden?
    \end{block}
    \uncover<2->{
    \begin{block}{Antwort:}
        Aufbau eines parallelen LC-Schwingkreises
    \end{block}}
    \uncover<3->{
    \begin{equation}
        f_{\symup{resonanz}} = \frac{1}{2\pi\sqrt{LC}}
    \end{equation}}
    \uncover<4->{
    \begin{equation}
        C =\frac{1}{L\left(2\pi f\right)^2}
    \end{equation}}

\end{frame}
